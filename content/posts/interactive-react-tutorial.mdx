## Custom Interactive Example

Let's build something from scratch:

<SandpackEditor
  template="react-ts"
  files={{
    '/App.tsx': 
      `
      import "./css-reset.css";
      import "./styles.css";

      const LENGTH = 3;

      export default function StackedComponent() {
        return (
          <div className="wrapper">
            {new Array(LENGTH).fill(0).map((_, i) => (
              <div className="card" key={i} style={{'--i': i}} />
            ))}
          </div>
        );
      }`,
    '/css-reset.css': `
      * {
        box-sizing: border-box;
        margin: 0;
        padding: 0;
      }

      html {
        color-scheme: dark;
        --gray1: #111111;
      --gray2: #191919;
      --gray3: #222222;
      --gray4: #2a2a2a;
      --gray5: #313131;
      --gray6: #3a3a3a;
      --gray7: #484848;
      --gray8: #606060;
      --gray9: #6e6e6e;
      --gray10: #7b7b7b;
      --gray11: #b4b4b4;
      --gray12: #eeeeee;
      --blue9: #0090ff;
      }

      body {
        height: 100vh;
        display: flex;
        align-items: center;
        justify-content: center;
        background: var(--gray1);
        font-family: system-ui, sans-serif;
      }`,
    '/styles.css': `
      .wrapper {
        display: grid;
      }

      .card {
        width: 356px;
        height: 74px;
        box-shadow: 0 4px 12px #0000001a;
        border: 1px solid #eeeeee;
        background: white;
        border-radius: 8px;
        grid-area: 1 / 1;
      }

      .card {
        --i: 0;
        --transformValue: calc(var(--i) * 10%);
        --scaleValue: calc(1 - (var(--i) * -0.05));

        transform: translateY(var(--transformValue)) scale(var(--scaleValue));
      }
    `

}}
height={450}
/>

## Vanilla JavaScript Example

Not everything needs React! Here's a vanilla JS example:

<SandpackEditor
  template="vanilla"
  files={{
    '/index.js': `const button = document.getElementById('btn');
const output = document.getElementById('output');
const colors = ['#3b82f6', '#ef4444', '#10b981', '#f59e0b', '#8b5cf6'];
let count = 0;

button.addEventListener('click', () => {
count++;
const randomColor = colors[Math.floor(Math.random() * colors.length)];
output.textContent = \`Clicked \${count} times\`;
output.style.color = randomColor;
});`,
    '/index.html': `<!DOCTYPE html>

<html>
<head>
  <title>Color Clicker</title>
  <style>
    body {
      font-family: system-ui, sans-serif;
      padding: 20px;
      text-align: center;
    }
    button {
      padding: 10px 20px;
      font-size: 16px;
      cursor: pointer;
      background: #3b82f6;
      color: white;
      border: none;
      border-radius: 4px;
    }
    #output {
      margin-top: 20px;
      font-size: 24px;
      font-weight: bold;
    }
  </style>
</head>
<body>
  <h1>Color Clicker</h1>
  <button id="btn">Click me!</button>
  <p id="output">Clicked 0 times</p>
  <script src="index.js"></script>
</body>
</html>`,
  }}
  height={400}
/>
